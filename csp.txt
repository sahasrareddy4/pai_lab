#include <iostream>
#include <vector>
using namespace std;

int N, M; 
vector<string> regions;
vector<string> colors;
vector<vector<int>> adjacency;
vector<int> assignedColor;
bool isSafe(int region, int c) {
    for (int i = 0; i < N; i++) {
        if (adjacency[region][i] == 1 && assignedColor[i] == c)
            return false;
    }
    return true;
}
bool solveCSP(int region) {
    if (region == N)
        return true;
    for (int c = 0; c < M; c++) {
        if (isSafe(region, c)) {
            assignedColor[region] = c;
            if (solveCSP(region + 1))
                return true;
            assignedColor[region] = -1; 
        }
    }
    return false;
}
int main() {
    cout << "Enter number of regions: ";
    cin >> N;
    cout << "Enter number of colors: ";
    cin >> M;
    regions.resize(N);
    colors.resize(M);
    adjacency.assign(N, vector<int>(N));
    assignedColor.assign(N, -1);
    cout << "\nEnter region names:\n";
    for (int i = 0; i < N; i++)
        cin >> regions[i];
    cout << "\nEnter color names:\n";
    for (int i = 0; i < M; i++)
        cin >> colors[i];
    cout << "\nEnter adjacency matrix (0 or 1):\n";
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            cin >> adjacency[i][j];
        }
    }
    if (solveCSP(0)) {
        cout << "\nColor Mapping:\n\n";
        for (int i = 0; i < N; i++) {
            cout << regions[i] << " => " << colors[assignedColor[i]] << endl;
        }
    } else {
        cout << "No solution exists!";
    }
    return 0;
}
