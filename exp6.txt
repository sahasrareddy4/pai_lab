#include <iostream>
using namespace std;

int MAX = 1000;
int MIN = -1000;

int minimax(int depth, int nodeIndex, bool maximizingPlayer, int values[], int alpha, int beta)
{
    if (depth == 3)
        return values[nodeIndex];

    if (maximizingPlayer)
    {
        int best = MIN;

        for (int i = 0; i < 2; i++)
        {
            int val = minimax(depth + 1, nodeIndex * 2 + i, false, values, alpha, beta);
            if (val > best)
                best = val;

            if (best > alpha)
                alpha = best;

            if (beta <= alpha)
                break;
        }
        return best;
    }
    else
    {
        int best = MAX;

        for (int i = 0; i < 2; i++)
        {
            int val = minimax(depth + 1, nodeIndex * 2 + i, true, values, alpha, beta);
            if (val < best)
                best = val;

            if (best < beta)
                beta = best;

            if (beta <= alpha)
                break;
        }
        return best;
    }
}

int main()
{
    int values[] = {3, 5, 6, 9, 1, 2, 0, -1};

    cout << "The optimal value is : ";
    cout << minimax(0, 0, true, values, MIN, MAX);

    return 0;
}
