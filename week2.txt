#include <iostream>
#include <queue>
using namespace std;

int main() {
    int A, B, T;
    cin >> A >> B >> T;

    bool *visited = new bool[(A + 1) * (B + 1)];

    for (int i = 0; i <= A; i++)
        for (int j = 0; j <= B; j++)
            visited[i * (B + 1) + j] = false;

    queue<int> qx, qy;

    qx.push(0);
    qy.push(0);
    visited[0] = true;

    while (!qx.empty()) {
        int x = qx.front(); qx.pop();
        int y = qy.front(); qy.pop();

        if (x == T || y == T) {
            cout << "Target reached at (" << x << ", " << y << ")\n";
            delete[] visited;
            return 0;
        }

        if (!visited[A * (B + 1) + y]) {
            visited[A * (B + 1) + y] = true;
            qx.push(A); qy.push(y);
        }

        if (!visited[x * (B + 1) + B]) {
            visited[x * (B + 1) + B] = true;
            qx.push(x); qy.push(B);
        }

        if (!visited[y]) {
            visited[y] = true;
            qx.push(0); qy.push(y);
        }

        if (!visited[x * (B + 1)]) {
            visited[x * (B + 1)] = true;
            qx.push(x); qy.push(0);
        }

        int pour = min(x, B - y);
        int nx = x - pour;
        int ny = y + pour;
        if (!visited[nx * (B + 1) + ny]) {
            visited[nx * (B + 1) + ny] = true;
            qx.push(nx); qy.push(ny);
        }

        pour = min(y, A - x);
        nx = x + pour;
        ny = y - pour;
        if (!visited[nx * (B + 1) + ny]) {
            visited[nx * (B + 1) + ny] = true;
            qx.push(nx); qy.push(ny);
        }
    }

    cout << "No solution exists\n";
    delete[] visited;
    return 0;
}
